\frametitle{Git: Branching}

\begin{block}{Branching}
Branching allows us to have several versions of file(s) at the same time.
\end{block}

\begin{multicols}{2}

\begin{lstlisting}[language=Bash]
git branch feature-1
git checkout feature-1
\end{lstlisting}
